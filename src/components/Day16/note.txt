‚≠ê STEP 1 ‚Äî Basic Setup (GET Request)
import axios from "axios";
import { useEffect, useState } from "react";

const CrudApp = () => {
  const [posts, setPosts] = useState([]);
  const [loading, setLoading] = useState(true);

  const API = "https://jsonplaceholder.typicode.com/posts";

  useEffect(() => {
    const fetchData = async () => {
      try {
        const res = await axios.get(API);
        setPosts(res.data.slice(0, 10)); // 10 posts only
      } catch (err) {
        console.error("Error fetching data", err);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  if (loading) return <h2 className="text-center mt-10">Loading...</h2>;

  return (
    <div className="p-6">
      <h1 className="text-3xl font-bold mb-4">CRUD App</h1>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {posts.map((post) => (
          <div key={post.id} className="p-4 bg-white shadow rounded">
            <h2 className="text-xl font-semibold">{post.title}</h2>
            <p className="text-gray-700">{post.body}</p>
          </div>
        ))}
      </div>
    </div>
  );
};

export default CrudApp;

‚≠ê EXPLANATION (NOTES)

üìù Notes & explanation (for your notes)

GET: axios.get(API) inside useEffect ‚Äî load on mount.

POST: axios.post(API, newPost) ‚Äî JSONPlaceholder returns created object; we add it to UI immediately.

DELETE: axios.delete(API/id) and filter it out from state. API on JSONPlaceholder doesn't persist but responds success ‚Äî UI still updated.

PUT: axios.put(API/id, updated) returns updated object; replace item in state using map.

editingId: used to toggle between Add and Edit mode.

UX: after edit/add we reset fields; we put new item at top ([res.data, ...posts]) for instant visibility.

Error handling: wrap in try/catch and show toast.
